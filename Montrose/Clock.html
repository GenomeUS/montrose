def initialize(opts = {})
  @options = Montrose::Options.merge(opts)
  @time = nil
  @every = @options.fetch(:every) { fail ConfigurationError, "Required option :every not provided" }
  @starts = @options.fetch(:starts)
  @interval = @options.fetch(:interval)
end
