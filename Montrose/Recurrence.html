def dump(obj)
  return nil if obj.nil?
  unless obj.is_a?(self)
    fail SerializationError,
      "Object was supposed to be a #{self}, but was a #{obj.class}. -- #{obj.inspect}"
  end

  JSON.dump(obj.to_hash)
end
