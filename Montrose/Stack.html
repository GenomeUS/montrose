def advance(time)
  yes, no = @stack.partition { |rule| rule.include?(time) }

  if no.empty?
    yes.all? { |rule| rule.advance!(time) } or return false
    puts time if ENV["DEBUG"]
    yield time if block_given?
    true
  else
    no.any?(&:continue?)
  end
end
